/**

 * Returns userId and cleanup function. Always call cleanup in finally block.

export async function seedFullAndLogin(
  page: Page,
  suffix: string,
  options: { activateCoach?: boolean; activateChef?: boolean } = {},
): Promise<SeedResult & { client: ReturnType<typeof createClient> }> {
  const { activateCoach = true, activateChef = true } = options;
  const { userId, email, password, cleanup } = await seedUser(suffix);

  // Create authenticated client for seeding data
  const client = createClient(SUPABASE_URL, ANON_KEY, {
    auth: { autoRefreshToken: false, persistSession: false },
  });
  await client.auth.signInWithPassword({ email, password });

  // Activate apps
  if (activateCoach) {
    await (client.schema('hub') as any).rpc('activate_app', { p_app_name: 'coachbyte' });
  }
  if (activateChef) {
    await (client.schema('hub') as any).rpc('activate_app', { p_app_name: 'chefbyte' });
  }

  // Log in via browser
  await page.goto('/login');
  await page.getByLabel('Email').fill(email);
  await page.getByLabel('Password').fill(password);
  await page.getByRole('button', { name: /sign in/i }).click();

  return { userId, email, password, cleanup, client };
